# 安装yasm插件

Yasm官网

# 下载
wget http://www.tortall.net/projects/yasm/releases/yasm-1.3.0.tar.gz
# 解压
tar -xvf yasm-1.3.0.tar.gz
# 执行安装
cd yasm-1.3.0/
./configure && make && make install

# 安装FFmpeg

# 下载

# 解压
tar -xvf ffmpeg-4.2.2.tar.gz
# 执行安装
cd ffmpeg-4.2.2/
./configure && make && make install
